// Глобальные переменные

// Функция для получения CSRF токена
function getCookie(name) {
    let cookieValue = null;
    if (document.cookie && document.cookie !== '') {
        const cookies = document.cookie.split(';');
        for (let i = 0; i < cookies.length; i++) {
            const cookie = cookies[i].trim();
            if (cookie.substring(0, name.length + 1) === (name + '=')) {
                cookieValue = decodeURIComponent(cookie.substring(name.length + 1));
                break;
            }
        }
    }
    return cookieValue;
}

// Закрытие модального окна
function closeModal(modalId) {
    document.getElementById(modalId).classList.remove('show');
    if (modalId === 'workModal') {
        document.getElementById('filePreviewSection').style.display = 'none';
        document.getElementById('filePreviewFrame').src = '';
        currentFileUrl = null;
    }
}

// Просмотр работы ученика
function decodeEscapedData(value) {
    if (!value) return '';
    return value
        .replace(/\\u([0-9a-fA-F]{4})/g, (_, hex) => String.fromCharCode(parseInt(hex, 16)))
        .replace(/\\n/g, '\n')
        .replace(/\\r/g, '\r')
        .replace(/\\t/g, '\t')
        .replace(/\\"/g, '"')
        .replace(/\\'/g, "'")
        .replace(/\\\\/g, '\\');
}

function formatFileSize(size) {
    if (!size || Number.isNaN(size)) return '';
    if (size < 1024) return size + ' Б';
    if (size < 1024 * 1024) return (size / 1024).toFixed(1) + ' КБ';
    return (size / (1024 * 1024)).toFixed(1) + ' МБ';
}

function fillSubmissionModal(data) {
    currentStudentId = data.studentId || null;
    currentSubmissionId = data.submissionId || null;
    selectedGrade = null;

    document.getElementById('modalStudentName').textContent = data.studentName || '';
    document.getElementById('submissionDate').textContent = data.submissionDate || '—';

    const submissionText = (data.submissionText || '').trim();
    if (submissionText) {
        document.getElementById('modalSubmissionText').textContent = submissionText;
        document.getElementById('modalSubmissionText').style.display = 'block';
        document.getElementById('noTextMessage').style.display = 'none';
    } else {
        document.getElementById('modalSubmissionText').textContent = '';
        document.getElementById('modalSubmissionText').style.display = 'none';
        document.getElementById('noTextMessage').style.display = 'block';
    }

    if (data.hasFile && data.fileUrl) {
        const rawName = data.fileName || '';
        const fileName = rawName.split('/').pop().split('\\').pop() || 'Файл';
        const fileSize = formatFileSize(data.fileSize);

        currentFileUrl = data.fileUrl;
        document.getElementById('fileSection').style.display = 'block';
        document.getElementById('fileLink').href = data.fileUrl + '?action=view';
        document.getElementById('fileDownloadLink').href = data.fileUrl + '?action=download';
        document.getElementById('fileName').textContent = fileName;
        document.getElementById('fileSize').textContent = fileSize;
    } else {
        currentFileUrl = null;
        document.getElementById('fileSection').style.display = 'none';
        document.getElementById('fileLink').href = '#';
        document.getElementById('fileDownloadLink').href = '#';
        document.getElementById('fileName').textContent = '';
        document.getElementById('fileSize').textContent = '';
    }

    document.getElementById('filePreviewSection').style.display = 'none';
    document.getElementById('filePreviewFrame').src = '';

    document.querySelectorAll('#workModal [data-grade-btn]').forEach(btn => {
        btn.classList.remove('selected');
    });

    const gradeValue = data.gradeValue ? parseFloat(data.gradeValue) : null;
    if (gradeValue) {
        selectedGrade = gradeValue;
        document.getElementById('gradeComment').value = data.gradeComment || '';
        const selectedBtn = document.querySelector('#workModal [data-grade-btn="' + gradeValue + '"]');
        if (selectedBtn) {
            selectedBtn.classList.add('selected');
        }
        document.getElementById('saveGradeBtn').disabled = false;
    } else {
        document.getElementById('gradeComment').value = '';
        document.getElementById('saveGradeBtn').disabled = true;
    }

    document.getElementById('workModal').classList.add('show');
}

// Просмотр работы ученика
function viewWork(buttonOrStudentId) {
    if (buttonOrStudentId && buttonOrStudentId.dataset) {
        const button = buttonOrStudentId;
        const fallbackData = {
            studentId: button.dataset.studentId || null,
            studentName: decodeEscapedData(button.dataset.studentName || ''),
            submissionId: button.dataset.submissionId || null,
            submissionText: decodeEscapedData(button.dataset.submissionText || ''),
            submissionDate: button.dataset.submissionDate || '—',
            hasFile: button.dataset.hasFile === '1',
            fileUrl: button.dataset.fileUrl || '',
            fileName: decodeEscapedData(button.dataset.fileName || ''),
            fileSize: Number(button.dataset.fileSize || 0),
            gradeValue: button.dataset.gradeValue || '',
            gradeComment: decodeEscapedData(button.dataset.gradeComment || ''),
        };

        fillSubmissionModal(fallbackData);

        const detailUrl = button.dataset.detailUrl || '';
        if (!detailUrl) {
            return;
        }

        fetch(detailUrl)
            .then(response => {
                if (!response.ok) {
                    throw new Error('HTTP error ' + response.status);
                }
                return response.json();
            })
            .then(data => {
                if (!data.success) {
                    throw new Error(data.error || 'Ошибка загрузки данных');
                }

                fillSubmissionModal({
                    studentId: data.student_id || fallbackData.studentId,
                    studentName: data.student_name || fallbackData.studentName,
                    submissionId: data.submission_id || fallbackData.submissionId,
                    submissionText: data.submission_text || '',
                    submissionDate: data.submission_date || fallbackData.submissionDate || '—',
                    hasFile: !!data.has_file,
                    fileUrl: data.file_url || '',
                    fileName: data.file_name || '',
                    fileSize: Number(data.file_size || 0),
                    gradeValue: data.has_grade ? data.grade_value : '',
                    gradeComment: data.grade_comment || '',
                });
            })
            .catch(error => {
                console.error('Ошибка загрузки данных работы:', error);
                showAlert('error', 'Не удалось обновить данные работы, показываю локальную версию');
            });
        return;
    }

    // Fallback for old onclick signature: viewWork(studentId, ...)
    const studentId = buttonOrStudentId;
    const row = document.querySelector(`tr[data-student-id="${studentId}"]`);
    const button = row ? row.querySelector('button[data-submission-id][onclick*="viewWork"]') : null;
    if (button) {
        viewWork(button);
        return;
    }

    showAlert('error', 'Не удалось открыть работу ученика');
}

function openSubmissionModalForStudent(studentId) {
    const row = document.querySelector(`tr[data-student-id="${studentId}"]`);
    const viewBtn = row ? row.querySelector('button[data-submission-id][onclick*="viewWork"]') : null;
    if (!viewBtn) {
        showAlert('error', 'Работа ученика не найдена');
        return;
    }
    viewWork(viewBtn);
}
// Выбор оценки в модальном окне
function selectGradeInModal(grade) {
    console.log('Выбрана оценка:', grade);
    selectedGrade = grade;
    
    document.querySelectorAll('#workModal [data-grade-btn]').forEach(btn => {
        btn.classList.remove('selected');
        if (btn.dataset.gradeBtn == grade) {
            btn.classList.add('selected');
        }
    });
    
    document.getElementById('saveGradeBtn').disabled = false;
}

// Сохранение оценки из модального окна
function saveGradeFromModal() {
    console.log('Сохранение оценки из модального окна');
    
    if (!selectedGrade) {
        showAlert('error', 'Выберите оценку');
        return;
    }
    
    if (!currentSubmissionId) {
        showAlert('error', 'Нет данных об отправке');
        return;
    }

// Оценка студента из таблицы
function gradeStudent(studentId, grade) {
    console.log('Оценка студента:', studentId, grade);
    
    const comment = '';
    
    const row = document.querySelector(`tr[data-student-id="${studentId}"]`);
    if (!row) {
        showAlert('error', 'Не найден ученик');
        return;
    }
    
    const viewBtn = row.querySelector('button[data-submission-id][onclick*="viewWork"]');
    if (!viewBtn) {
        showAlert('error', 'Нет данных о работе');
        return;
    }
    
    let submissionId = viewBtn.getAttribute('data-submission-id');
    
    if (!submissionId) {
        const onclickText = viewBtn.getAttribute('onclick') || '';
        const match = onclickText.match(/viewWork\(\s*['"]?\d+['"]?\s*,\s*['"][^'"]*['"]\s*,\s*['"]?(\d+)['"]?/);
        if (match && match.length >= 2) {
            submissionId = match[1];
        }
    }
    
    if (!submissionId) {
        showAlert('error', 'Не удалось получить данные работы');
        return;
    }
    
    saveGrade(submissionId, grade, comment, false, studentId);
}

// Сохранение оценки
function saveGrade(submissionId, grade, comment, shouldCloseModal = false, studentIdForUpdate = null) {
    console.log('Сохранение оценки:', submissionId, grade);
    
    const formData = new FormData();
    formData.append('value', grade);
    formData.append('comment', comment);
    
    fetch(`{% url 'teacher_portal:grade_submission' 0 %}`.replace('0', submissionId), {
        method: 'POST',
        headers: {
            'X-CSRFToken': getCookie('csrftoken'),
        },
        body: formData
    })
    .then(response => {
        if (!response.ok) {
            return response.json().then(err => { throw new Error(err.error || 'HTTP error ' + response.status); });
        }
        return response.json();
    })
    .then(data => {
        if (data.success) {
            const targetStudentId = studentIdForUpdate || currentStudentId;
            if (targetStudentId) {
                updateStudentGrade(targetStudentId, grade);
                const row = document.querySelector(`tr[data-student-id="${targetStudentId}"]`);
                const viewBtn = row ? row.querySelector('button[data-submission-id][onclick*="viewWork"]') : null;
                if (viewBtn) {
                    viewBtn.dataset.gradeValue = String(grade);
                    viewBtn.dataset.gradeComment = comment || '';
                }
            }
            showAlert('success', data.message || 'Оценка сохранена');
            
            if (shouldCloseModal) {
                closeModal('workModal');
            }
        } else {
            showAlert('error', data.error || 'Ошибка при сохранении оценки');
        }
    })
    .catch(error => {
        console.error('Error:', error);
        showAlert('error', 'Ошибка: ' + error.message);
    });
}

    `;
}

// Показ уведомлений
function showAlert(type, message) {
    const alertDiv = document.createElement('div');
    alertDiv.className = `alert alert-${type}`;
    const isError = type === 'error';
    alertDiv.style.cssText = `position: fixed; top: 20px; right: 20px; z-index: 9999; padding: 12px 16px; border-radius: 8px; background: ${isError ? 'var(--danger-light)' : 'var(--success-light)'}; border: 1px solid ${isError ? 'var(--danger)' : 'var(--success)'}; color: ${isError ? 'var(--danger)' : 'var(--success)'};`;
    alertDiv.innerHTML = `
        ${message}
        <button type="button" style="background: none; border: none; color: inherit; margin-left: 12px; cursor: pointer;" onclick="this.parentElement.remove()">×</button>
    `;
    
    document.body.appendChild(alertDiv);
    
    setTimeout(() => {
        if (alertDiv.parentElement) {
            alertDiv.remove();
        }
    }, 3000);
}
